Question 3. Function to Format Followers Count 

function formatFollowersCount(count)
{
    if (count >= 1000000)
    {
        return (count / 1000000) + "M";
    }
    else if (count >= 1000)
    {
        return (count / 1000) + "K";
    }
    else
    {
        return count;
    }
}

console.log(formatFollowersCount(1500));
console.log(formatFollowersCount(1200000));
